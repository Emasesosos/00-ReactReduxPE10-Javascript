02-Scope

Scope con var

var musica = 'Rock';

if (musica) {

var musica = 'Grunge';

console.log('Dentro del IF: ', musica);

}

console.log('Fuera del IF: ', musica);

Scope con let

let musicaL = 'Rock';

if (musicaL) {

let musicaL = 'Grunge';

console.log('Dentro del IF: ', musicaL);

}

console.log('Fuera del IF', musicaL);

03-Template Strings

Concatenar con Javascript

const nombre = 'Jesús';

const trabajo = 'Ingeniero de Software';

console.log('Nombre: ' + nombre + ', Trabajo: ' + trabajo);

Concatenar con Template Strings

const nombre = 'Jesús';

const trabajo = 'Ingeniero de Software';

console.log(`Nombre: ${nombre}, Trabajo: ${trabajo}`);
